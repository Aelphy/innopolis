	\section{Question 2}
		\begin{equation}
			L_1(a, b) = |a_x - b_x| + |a_y - b_y|
		\end{equation}
		
		\begin{equation}
			L_2(a, b) = \sqrt{(a_x - b_x)^2 + (a_y - b_y)^2}
		\end{equation}
		
		\begin{equation}
			L_{\infty} (a, b) = \max \left\{|a_x - b_x|, |a_y- b_y||\right\}
		\end{equation}
		
		8-connected distance equals to $L_{\infty}$ \cite{morse1998lecture}
		
		Hence: 
		\begin{equation}
			L_1(S, T_1) = 10
		\end{equation}
		
		\begin{equation}
			L_1(S, T_2) = 8
		\end{equation}
		
		\begin{equation}
			L_1(S, T_3) = 16
		\end{equation}
		
		\begin{equation}
			L_2(S, T_1) = \sqrt{68}
		\end{equation}
		
		\begin{equation}
			L_2(S, T_2) = \sqrt{40}
		\end{equation}
		
		\begin{equation}
			L_2(S, T_3) = \sqrt{136}
		\end{equation}
		
		\begin{equation}
			L_3(S, T_1) = 8
		\end{equation}
		
		\begin{equation}
			L_3(S, T_2) = 6
		\end{equation}
		
		\begin{equation}
			L_3(S, T_3) = 10
		\end{equation}
		
		Thus, in all distance metrics the closest one to S id $T_2$
